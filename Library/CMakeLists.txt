FILE(GLOB LIBRARY_SOURCES *.cc)

SET(LIBRARY_NAME MyClass)
ADD_LIBRARY(${LIBRARY_NAME} SHARED ${LIBRARY_SOURCES})

SET_TARGET_PROPERTIES(${LIBRARY_NAME} PROPERTIES LABELS "LIB")
SET_TARGET_PROPERTIES(${LIBRARY_NAME} PROPERTIES PREFIX "")

SET(PINNACLE_LIBRARIES ${LIBRARY_NAME} CACHE INTERNAL "")

SET_TARGET_PROPERTIES(${LIBRARY_NAME} 
  PROPERTIES
  LIBRARY_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/build/bin"
)

INSTALL(TARGETS ${LIBRARY_NAME}
  RUNTIME DESTINATION lib
  LIBRARY DESTINATION lib
  ARCHIVE DESTINATION lib
)
